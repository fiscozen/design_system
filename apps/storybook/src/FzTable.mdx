# FzTable

A feature-rich data table component with sorting, filtering, pagination, row selection, accordion sub-rows, and programmatic state control. Renders rows using a CSS grid layout with support for multiple variants.

## Features

- **Multiple Variants**: `normal`, `accordion`, `list`, and `radio` display modes
- **Column Sorting**: Orderable columns with ascending/descending/none cycle
- **Pagination**: Built-in pagination with configurable page intervals
- **Row Selection**: Checkbox-based multi-select or radio single-select
- **Accordion Sub-Rows**: Expandable parent rows with nested sub-row data, with programmatic open/close control via `v-model:openRowIds`
- **Search**: Integrated fulltext search input with debounced updates
- **Filters**: Side panel with customizable filter slots
- **Actions**: Per-row action dropdown menus, supporting static or dynamic (function-based) actions
- **Loading State**: Overlay spinner during async operations
- **Fullscreen Mode**: Optional fullscreen toggle
- **Responsive**: Adapts layout for small screens
- **Accessibility**: Semantic `role="table"` structure with `aria-sort`, `aria-rowcount`, and `aria-colcount`

## Installation

```bash
npm install @fiscozen/table
```

## Basic Usage

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { FzTable } from '@fiscozen/table'
import { FzColumn } from '@fiscozen/simple-table'

const data = ref([
  { name: 'Alice', email: 'alice@example.com' },
  { name: 'Bob', email: 'bob@example.com' },
])
</script>

<template>
  <FzTable v-model="data" title="Users">
    <FzColumn header="Name" field="name" />
    <FzColumn header="Email" field="email" />
  </FzTable>
</template>
```

## Examples

### With Sorting and Pagination

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { FzTable, FzOrdering } from '@fiscozen/table'
import { FzColumn } from '@fiscozen/simple-table'

const data = ref([...])
const activePage = ref(0)
const ordering = ref<Record<string, FzOrdering>>({
  name: { orderable: true, direction: 'none' },
  email: { orderable: true, direction: 'none' },
})
</script>

<template>
  <FzTable
    v-model="data"
    v-model:ordering="ordering"
    v-model:activePage="activePage"
    :pages="10"
    :internalOrdering="true"
    title="Users"
  >
    <FzColumn header="Name" field="name" />
    <FzColumn header="Email" field="email" />
  </FzTable>
</template>
```

### Accordion with Programmatic Control

Use `v-model:openRowIds` to control which accordion rows are expanded. The Set contains row IDs (from `row.id` or the row's array index).

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { FzTable } from '@fiscozen/table'
import { FzColumn } from '@fiscozen/simple-table'

const data = ref([
  {
    id: 'team-a',
    name: 'Team A',
    email: 'team-a@example.com',
    subRows: [
      { name: 'Alice', email: 'alice@example.com' },
      { name: 'Bob', email: 'bob@example.com' },
    ],
  },
  {
    id: 'team-b',
    name: 'Team B',
    email: 'team-b@example.com',
    subRows: [
      { name: 'Charlie', email: 'charlie@example.com' },
    ],
  },
])

// Pre-expand Team A
const openRowIds = ref(new Set<string | number>(['team-a']))

const openAll = () => {
  openRowIds.value = new Set(data.value.map(r => r.id))
}

const closeAll = () => {
  openRowIds.value = new Set()
}
</script>

<template>
  <button @click="openAll">Open All</button>
  <button @click="closeAll">Close All</button>

  <FzTable
    v-model="data"
    v-model:openRowIds="openRowIds"
    variant="accordion"
    title="Teams"
  >
    <FzColumn header="Name" field="name" />
    <FzColumn header="Email" field="email" />
  </FzTable>
</template>
```

When no `v-model:openRowIds` is provided, the component manages the open state internally (clicking a row toggles it).

### With Search and Filters

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { FzTable } from '@fiscozen/table'
import { FzColumn } from '@fiscozen/simple-table'

const data = ref([...])
const searchTerm = ref('')
const hasActiveFilters = ref(false)

const handleSearch = (term: string | undefined) => {
  searchTerm.value = term || ''
  // Filter data based on searchTerm...
}
</script>

<template>
  <FzTable
    v-model="data"
    :searchable="true"
    :hasActiveFilters="hasActiveFilters"
    title="Users"
    @update:searchTerm="handleSearch"
    @fztable:emptyFilters="hasActiveFilters = false"
  >
    <FzColumn header="Name" field="name" :filterable="true" />
    <FzColumn header="Email" field="email" />
    <template #filter-name>
      <!-- Custom filter UI for the Name column -->
    </template>
  </FzTable>
</template>
```

### Dynamic Actions per Row

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { FzTable } from '@fiscozen/table'
import { FzColumn } from '@fiscozen/simple-table'

const data = ref([
  { name: 'Alice', role: 'admin' },
  { name: 'Bob', role: 'user' },
])

const getActions = (row: any) => ({
  items: [
    { type: 'button' as const, label: 'Edit' },
    ...(row.role === 'admin'
      ? [{ type: 'button' as const, label: 'Manage permissions' }]
      : []),
  ],
})
</script>

<template>
  <FzTable v-model="data" :actions="getActions" title="Users">
    <FzColumn header="Name" field="name" />
    <FzColumn header="Role" field="role" />
  </FzTable>
</template>
```

## Accessibility

FzTable follows WCAG 2.1 AA guidelines:

- **Semantic Structure**: Uses `role="table"`, `role="columnheader"`, and `role="cell"` for assistive technology
- **ARIA Attributes**: `aria-rowcount`, `aria-colcount`, `aria-rowindex`, and `aria-sort` on sortable columns
- **Keyboard Navigation**: Column headers are clickable to toggle sort order
- **Selection**: Checkbox and radio inputs provide accessible selection patterns
- **Fullscreen**: Escape key exits fullscreen mode
